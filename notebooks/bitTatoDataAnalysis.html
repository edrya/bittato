import pandas as pd
import numpy  as np
import matplotlib.pyplot as plt
import seaborn as sns
import json

pd.options.display.float_format = '{:,.2f}'.format

# Loading Datasets
potatoes = pd.read_csv('data/PotatoesInfo.csv')
clients = pd.read_csv('data/ClientInfo.csv')
positions = pd.read_csv('data/PotatoesPositions.csv', parse_dates=['Date'])

# Data Cleaning and Preparation
positions = positions.replace('[NULL]', np.nan)
positions = positions.dropna()

# Create new columns 
positions['Quantity'] = positions['Quantity'].astype('float')
positions['Market Value'] = positions['Price'] * positions['Quantity']
clients['Full Name'] = clients['First Name'] + ' ' + clients['Last Name']

# Combining and Merging Datasets
positions = pd.merge(positions, potatoes, on=['Product ID'])
positions = pd.merge(positions, clients, on=['Client ID'])

positions = positions.set_index('Date').sort_index()
positions['Weekday Name'] = positions.index.weekday_name
